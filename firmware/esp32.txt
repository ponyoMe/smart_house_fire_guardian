#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include <time.h>

// ===================== WIFI =====================
const char* ssid = "Beeline";
const char* password = "05050606";

// ===================== MQTT =====================
const char* mqttServer = "a4af6d33f79a4077860f7a5fed91b077.s1.eu.hivemq.cloud";
const int mqttPort = 8883;
const char* mqttUser = "iot_smart_home";
const char* mqttPassword = "Password1";

// ===================== MQTT CONTROL SUB =====================
const char* mqttControlSubscribe = "home/+/+/control";

// ===================== UART =====================
#define RXD2 16
#define TXD2 17

WiFiClientSecure espClient;
PubSubClient client(espClient);

String serialBuffer = "";

unsigned long lastHeartbeat = 0;
const unsigned long heartbeatInterval = 30000;

// ===================== NTP =====================
void setupNTP() {
  configTime(0, 0, "pool.ntp.org", "time.nist.gov");

  Serial.print("Waiting for NTP sync");
  time_t now = time(nullptr);
  int attempts = 0;

  while (now < 1000000000UL && attempts < 20) {
    delay(500);
    Serial.print(".");
    now = time(nullptr);
    attempts++;
  }

  Serial.println();
  if (now >= 1000000000UL) Serial.println("NTP synced");
  else Serial.println("NTP failed, using millis fallback");
}

unsigned long getTimestamp() {
  time_t now = time(nullptr);
  return (now >= 1000000000UL) ? (unsigned long)now : millis();
}

// ===================== WIFI =====================
void setupWiFi() {
  Serial.println();
  Serial.print("Connecting to WiFi: ");
  Serial.println(ssid);

  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);

  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 40) {
    delay(500);
    Serial.print(".");
    attempts++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println();
    Serial.print("WiFi connected, IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println();
    Serial.println("WiFi failed, restarting...");
    ESP.restart();
  }
}

// ===================== TOPIC HELPERS =====================
String getRoomFromDeviceId(const String &deviceId) {
  if (deviceId == "esp32") return "living"; // change if esp32 belongs to another room

  int idx = deviceId.indexOf('_');
  if (idx == -1) return "unknown";

  return deviceId.substring(0, idx);
}

String buildTopic(const String &deviceId, const String &msgType) {
  String room = getRoomFromDeviceId(deviceId);
  return "home/" + room + "/" + deviceId + "/" + msgType;
}

// ===================== JSON HELPERS =====================
String extractJsonStringValue(const String &json, const String &key) {
  String pattern = "\"" + key + "\":\"";
  int keyIdx = json.indexOf(pattern);
  if (keyIdx == -1) return "";

  int start = keyIdx + pattern.length();
  int end = json.indexOf("\"", start);
  if (end == -1) return "";

  return json.substring(start, end);
}

String addTimestamp(String msg) {
  int pos = msg.indexOf('{');
  if (pos == -1) return msg;

  // avoid adding ts twice
  if (msg.indexOf("\"ts\":") != -1) return msg;

  return msg.substring(0, pos + 1)
       + "\"ts\":" + String(getTimestamp()) + ","
       + msg.substring(pos + 1);
}

// ===================== PUBLISH =====================
void publishJSON(const char* topic, const String &payload) {
  bool ok = client.publish(topic, payload.c_str());
  Serial.print("Publish to ");
  Serial.print(topic);
  Serial.print(" -> ");
  Serial.println(ok ? "OK" : "FAILED");
  Serial.println(payload);
}

void publishStatus(const String &deviceId, const String &status, const String &message) {
  String payload =
    "{\"deviceId\":\"" + deviceId + "\","
    "\"type\":\"status\","
    "\"ts\":" + String(getTimestamp()) + ","
    "\"data\":{"
      "\"status\":\"" + status + "\","
      "\"message\":\"" + message + "\""
    "}}";

  String topic = buildTopic(deviceId, "status");
  publishJSON(topic.c_str(), payload);
}

// ===================== MQTT CALLBACK =====================
void mqttCallback(char* topic, byte* payload, unsigned int length) {
  String incoming = "";
  for (unsigned int i = 0; i < length; i++) {
    incoming += (char)payload[i];
  }

  Serial.println("MQTT in [" + String(topic) + "]: " + incoming);

  // expected JSON like:
  // {"command":"POWER_ON"}
  String cmd = incoming;
  int cmdIdx = incoming.indexOf("\"command\":\"");

  if (cmdIdx != -1) {
    int start = cmdIdx + 11;
    int end = incoming.indexOf("\"", start);
    if (end != -1) {
      cmd = incoming.substring(start, end);
    }
  }

  Serial2.println(cmd);
  Serial.println("Forwarded to Mega: " + cmd);
}

// ===================== MQTT RECONNECT =====================
void reconnectMQTT() {
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");

    String clientId = "ESP32Client-" + String((uint32_t)ESP.getEfuseMac(), HEX);

    if (client.connect(clientId.c_str(), mqttUser, mqttPassword)) {
      Serial.println("connected");

      client.subscribe(mqttControlSubscribe);
      Serial.print("Subscribed to: ");
      Serial.println(mqttControlSubscribe);

      publishStatus("esp32", "ONLINE", "ESP32 connected to broker");
    } else {
      Serial.print("failed rc=");
      Serial.print(client.state());
      Serial.println(", retry in 5s");
      delay(5000);
    }
  }
}

// ===================== PROCESS MESSAGE FROM MEGA =====================
void processMegaMessage(String msg) {
  msg.trim();
  if (msg.length() == 0) return;

  Serial.println("From Mega: " + msg);

  if (!msg.startsWith("{")) {
    Serial.println("Invalid message (not JSON)");
    return;
  }

  String enriched = addTimestamp(msg);

  String deviceId = extractJsonStringValue(enriched, "deviceId");
  if (deviceId.length() == 0) {
    Serial.println("No deviceId in message");
    return;
  }

  if (enriched.indexOf("\"type\":\"telemetry\"") != -1) {
    String topic = buildTopic(deviceId, "telemetry");
    publishJSON(topic.c_str(), enriched);
  }
  else if (enriched.indexOf("\"type\":\"status\"") != -1) {
    String topic = buildTopic(deviceId, "status");
    publishJSON(topic.c_str(), enriched);
  }
  else if (enriched.indexOf("\"type\":\"alert\"") != -1) {
    String topic = buildTopic(deviceId, "status");
    publishJSON(topic.c_str(), enriched);
  }
  else {
    Serial.println("Unknown message type from Mega");
  }
}

// ===================== SETUP =====================
void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial2.begin(115200, SERIAL_8N1, RXD2, TXD2);

  espClient.setInsecure();

  setupWiFi();
  setupNTP();

  client.setServer(mqttServer, mqttPort);
  client.setCallback(mqttCallback);
  client.setKeepAlive(60);
  client.setBufferSize(2048);

  Serial.println("ESP32 ready");
}

// ===================== LOOP =====================
void loop() {
  if (WiFi.status() != WL_CONNECTED) {
    setupWiFi();
  }

  if (!client.connected()) {
    reconnectMQTT();
  }

  client.loop();

  while (Serial2.available()) {
    char c = Serial2.read();

    if (c == '\n') {
      processMegaMessage(serialBuffer);
      serialBuffer = "";
    } else if (c != '\r') {
      serialBuffer += c;
    }
  }

  if (millis() - lastHeartbeat >= heartbeatInterval) {
    lastHeartbeat = millis();
    publishStatus("esp32", "HEARTBEAT", "ESP32 alive");
  }
}
